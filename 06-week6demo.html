5 1.Word Embedding Analysis

library(tidytext)
library(widyr)
library(Matrix)
library(RSpectra)

# create skipgrams
tidy_skipgrams <- emails %>%
  unnest_tokens(ngram, value, token = "ngrams", n = 6) %>%
  mutate(ngramID = row_number()) %>%
  tidyr::unite(skipgramID, ngramID) %>%
  unnest_tokens(word, ngram)

# co-occurrence
skipgram_probs <- tidy_skipgrams %>%
  pairwise_count(word, skipgramID, diag = TRUE, sort = TRUE) %>%
  mutate(p = n / sum(n))

# unigram
unigram_probs <- emails %>%
  unnest_tokens(word, value) %>%
  count(word) %>%
  mutate(p = n / sum(n))

# PMI
normalized_prob <- skipgram_probs %>%
  filter(n > 5) %>%
  rename(word1 = item1, word2 = item2) %>%
  left_join(unigram_probs %>% select(word1 = word, p1 = p), by = "word1") %>%
  left_join(unigram_probs %>% select(word2 = word, p2 = p), by = "word2") %>%
  mutate(p_together = p / p1 / p2)

# matrix
pmi_matrix <- normalized_prob %>%
  mutate(pmi = log10(p_together)) %>%
  cast_sparse(word1, word2, pmi)

pmi_matrix@x[is.na(pmi_matrix@x)] <- 0

# SVD
pmi_svd <- svds(pmi_matrix, 50)

word_vectors <- pmi_svd$u
rownames(word_vectors) <- rownames(pmi_matrix)

At this stage, I apply Singular Value Decomposition (SVD) to the PMI matrix in order to generate word embeddings. The PMI matrix captures how strongly words co-occur relative to chance, and SVD reduces this high-dimensional matrix into a lower-dimensional space. The resulting object, word_vectors, represents each word as a numeric vector. Words that appear in similar contexts will have similar vector representations. This allows us to measure semantic similarity between words using vector operations, such as the dot product. In this model, each word is represented as a 50-dimensional vector. This dimensionality is a design choice that balances capturing semantic variation while keeping the model computationally efficient.

dim(word_vectors)
## [1] 844  50

The resulting word embedding matrix contains 844 unique words, each represented as a 50-dimensional vector. This means that every word in the corpus is mapped onto a point in a 50-dimensional semantic space. Words that appear in similar contexts are expected to have similar vector representations, allowing us to measure semantic similarity between words.

nearest_words <- function(word_vectors, word){
  selected_vector = word_vectors[word,]
  mult = as.data.frame(word_vectors %*% selected_vector)

  mult %>%

tidy_skipgrams <- emails %>%

  pairwise_count(word, skipgramID, diag = TRUE, sort = TRUE) %>%
  mutate(p = n / sum(n))

# unigram
unigram_probs <- emails %>%
  unnest_tokens(word, value) %>%
  count(word) %>%
  mutate(p = n / sum(n))

# PMI
normalized_prob <- skipgram_probs %>%
  filter(n > 5) %>%
  rename(word1 = item1, word2 = item2) %>%
  left_join(unigram_probs %>% select(word1 = word, p1 = p), by = "word1") %>%
  left_join(unigram_probs %>% select(word2 = word, p2 = p), by = "word2") %>%
  mutate(p_together = p / p1 / p2)

# matrix
pmi_matrix <- normalized_prob %>%
  mutate(pmi = log10(p_together)) %>%
  cast_sparse(word1, word2, pmi)

pmi_matrix@x[is.na(pmi_matrix@x)] <- 0

# SVD
